04-ES6

ES 6

1. let && const

let 定义变量，const 定义常量
let 不支持重复定义

let 只在块级作用域中起作用

不能提升作用域

console.log(foo);
var foo = 1;

编译后为：
var foo;
console.log(foo)
foo = 1

使用 let

console.log(foo);
var foo = 1;

程序报错，提示 foo 未定义

2. 箭头函数

2.1 参数 => 表达式/语句

let val = 2;
let double = x => 2 * x;
let treble = x => {
    return 3 * x;
}
console.log(double(val));
console.log(treble(val));

2.2 继承外层作用域，没有独立作用域

var obj {
    commmonFn : function(){
        console.log(this);
    },
    arrowFn : () => {
        console.log(this);
    }
}

obj.commonFn(); // this 指向 obj 对象,输出是 obj 对象的内容
obj.arrowFn();  // arrowFn 是没有作用域的，所以其与 obj 共享一个作用域，obj 是 window，所以 arrowFn 也输出是 window 对象的内容

2.3 不能用作构造函数 

// 不报错
let Animal = function(){}
let animal = new Animal();

// 报错
let Dog = () => {}
let dog = new Dog();

2.4 没有 prototype 属性 

let commonFn = function() {}
let arrowFn = () => {}

console.log(commonFn.prototype);
console.log(arrowFn.prototype);

3. 模板字符串

3.1 反引号标识 ``，支持多行字符串

let str = `
    <div>
        <h1 class="title>>123</h1>
    </div>
`;
document.querySelector('body').innerText = str;

3.2 支持变量和表达式

let name = "Javaliu";
let str = `
    <div>
        <h1 class="title>>${name}</h1>
    </div>
`;
document.querySelector('body').innerText = str;

3.3 嵌套函数和循环嵌套的使用

let getName = () => 'abc';
let str = `
    <div>
        <h1 class="title>>${getName()}</h1>
    </div>
`;
document.querySelector('body').innerText = str;

循环嵌套的例子

let names = ['javaliu', 'lili', 'lucy'];
let str = `
    <ul>
        ${
            names.map(name => `<li>Hi, I am ${name}</li>`).join('');
        }
    </ul>
`;
document.querySelector('body').innerText = str;
